            const loadKingJohn = async () => {
                try {
                    console.log('🔄 Loading King John...');
                    const response = await fetch('Public/Data/king_john.json?v=' + Date.now());
                    
                    if (response.ok) {
                        const kingjohnData = await response.json();
                        window.kingjohnData = kingjohnData;
                        
                        plays['kingjohn'] = {
                            name: 'King John',
                            originalFileName: 'kingjohn',
                            characters: [],
                            acts: {},
                            scenes: {}
                        };
                        
                        if (kingjohnData) {
                            let sceneCount = 0;
                            let lineCount = 0;
                            Object.keys(kingjohnData).forEach(originalSceneKey => {
                                const sceneData = kingjohnData[originalSceneKey];
                                if (sceneData && typeof sceneData === 'object') {
                                    const sceneContent = [];
                                    Object.keys(sceneData).forEach(lineNum => {
                                        const lineData = sceneData[lineNum];
                                        if (!lineData) return;
                                        const text = (typeof lineData === 'string') ? lineData : (lineData.play ?? lineData.text ?? lineData.line ?? null);
                                        if (text) {
                                            sceneContent.push(text);
                                            lineCount++;
                                        }
                                    });
                                    if (sceneContent.length > 0) {
                                        const optimizedContent = optimizeSpeakerNames(sceneContent);
                                        const normalizedKey = normalizeSceneKey(originalSceneKey);
                                        plays['kingjohn'].scenes[normalizedKey] = optimizedContent;
                                        sceneCount++;
                                        const actName = parseActFromScene(normalizedKey);
                                        if (actName) {
                                            if (!plays['kingjohn'].acts[actName]) {
                                                plays['kingjohn'].acts[actName] = [];
                                            }
                                            if (!plays['kingjohn'].acts[actName].includes(normalizedKey)) {
                                                plays['kingjohn'].acts[actName].push(normalizedKey);
                                            }
                                        }
                                    }
                                }
                            });
                            console.log(`✅ King John parsed: ${sceneCount} scenes, ${lineCount} lines`);
                        }
                        return 'kingjohn';
                    } else {
                        console.warn('❌ Could not load King John data - Status:', response.status);
                        return null;
                    }
                } catch (error) {
                    console.error('❌ Error loading King John:', error);
                    return null;
                }
            };
